## 요구사항 2: ESP ↔ Orbital 토글 시각화

### 문제 핵심

`showControls(mode)`가 현재 모드와 다른 쪽의 컨트롤을 완전히 숨겨버려서, 사용자가 ESP를 보다가 Orbital로 전환하거나 그 반대를 할 수 없습니다.

### viewer.js — 수정

```javascript
// ═══════════════════════════════════════════════════════════
// 요구사항 2: ESP ↔ Orbital 자유 전환 토글
// ═══════════════════════════════════════════════════════════

/**
 * showControls를 개편하여, 결과에 두 종류 데이터가
 * 모두 있으면 두 버튼 모두 노출 + 활성 상태 표시
 */
function showControls(activeMode) {
  if ($empty) $empty.hidden = true;
  if ($controls) $controls.hidden = false;

  var result = state.result || {};
  var hasOrbital = !!(
    result.orbital_cube_b64 ||
    result.orbital_cube ||
    (result.orbitals && result.orbitals.length)
  );
  var hasESP = !!(result.esp_cube_b64 || result.esp_cube);

  /* ── 오비탈 컨트롤 그룹 ── */
  if ($grpOrbital) {
    $grpOrbital.hidden = !hasOrbital;
  }

  /* ── ESP 컨트롤 그룹 ── */
  if ($grpESP) {
    $grpESP.hidden = !hasESP;
  }

  /* ── 투명도/isovalue 슬라이더: 어느 모드든 표면이 있으면 표시 ── */
  if ($grpOpacity) {
    $grpOpacity.hidden = !(hasOrbital || hasESP);
  }

  /* ── 토글 버튼 그룹: 두 데이터 모두 있을 때만 노출 ── */
  if ($vizModeToggle) {
    $vizModeToggle.hidden = !(hasOrbital && hasESP);
  }

  /* 활성 모드 버튼 하이라이트 */
  updateToggleHighlight(activeMode);
}

/**
 * 토글 버튼의 활성(active) 클래스를 전환
 */
function updateToggleHighlight(mode) {
  var $btnOrb = document.getElementById("btnModeOrbital");
  var $btnEsp = document.getElementById("btnModeESP");
  if ($btnOrb) {
    $btnOrb.classList.toggle("active", mode === "orbital");
  }
  if ($btnEsp) {
    $btnEsp.classList.toggle("active", mode === "esp");
  }
}

/**
 * 모드 전환 핸들러 — 구조(XYZ)는 그대로 두고 표면만 교체
 */
function switchVizMode(newMode) {
  if (!state.result || state.mode === newMode) return;

  var result = state.result;
  state.mode = newMode;

  /* 표면만 지우고 분자 모델은 유지 */
  if (state.viewer) {
    state.viewer.removeAllSurfaces();
  }

  if (newMode === "orbital") {
    var cubeData =
      result.orbital_cube ||
      (result.orbital_cube_b64 ? atob(result.orbital_cube_b64) : null);
    if (cubeData) {
      renderOrbitalSurface(cubeData, state.isovalue, state.opacity);
    }
  } else if (newMode === "esp") {
    var espData =
      result.esp_cube ||
      (result.esp_cube_b64 ? atob(result.esp_cube_b64) : null);
    var denData =
      result.density_cube ||
      (result.density_cube_b64 ? atob(result.density_cube_b64) : null);
    if (espData) {
      renderESPSurface(espData, denData, {
        density_iso: state.espDensityIso,
        esp_min: result.esp_min,
        esp_max: result.esp_max,
      });
    }
  }

  showControls(newMode);
  saveViewerSnapshot();
}
```

## 요구사항 3: Geometry Optimization Trajectory 시각화

### 3‑A. 백엔드 — `pyscf_runner.py` 수정

```python
# ═══════════════════════════════════════════════════════════
# 요구사항 3: 최적화 궤적(Trajectory) 수집
# ═══════════════════════════════════════════════════════════

def run_geometry_optimization(
    mol_input, method="RHF", basis="sto-3g",
    progress_callback=None, **kwargs
):
    """Geometry optimization with per‑step trajectory capture."""

    # ... (기존 mol/mf 생성 코드 동일) ...

    trajectory = []  # ★ 궤적 수집용 리스트

    def _geomopt_callback(envs):
        """geometric 최적화의 매 스텝 콜백.
        envs dict에 현재 좌표와 에너지가 들어있다."""
        try:
            mol_current = envs.get("mol", None)
            e_current = envs.get("e_tot", None)
            grad_norm = None

            # gradients가 있으면 norm 계산
            g = envs.get("gradients", None)
            if g is not None:
                import numpy as np
                grad_norm = float(np.linalg.norm(g))

            step_num = len(trajectory) + 1

            # 현재 구조를 XYZ 문자열로 변환
            xyz_string = _mol_to_xyz_string(mol_current) if mol_current else None

            step_data = {
                "step": step_num,
                "energy_hartree": float(e_current) if e_current is not None else None,
                "grad_norm": grad_norm,
                "xyz": xyz_string,
            }
            trajectory.append(step_data)

            # 진행률 보고
            if progress_callback:
                # 최대 50스텝 가정, 0.3~0.85 구간에 매핑
                frac = min(0.3 + (step_num / 50) * 0.55, 0.85)
                _emit_progress(
                    progress_callback, frac, "optimize",
                    f"Opt step {step_num}: E={e_current:.8f} Ha, "
                    f"|grad|={grad_norm:.6f}" if grad_norm else
                    f"Opt step {step_num}: E={e_current:.8f} Ha"
                )
        except Exception:
            pass  # 콜백 실패가 최적화 자체를 중단시키면 안 됨

    # ── 최적화 실행 ──
    _emit_progress(progress_callback, 0.30, "optimize",
                   "Starting geometry optimization")

    try:
        from pyscf.geomopt.geometric_solver import optimize as geometric_optimize

        opt_mol = geometric_optimize(
            mf0,
            callback=_geomopt_callback,
            maxsteps=kwargs.get("max_steps", 100),
        )
        optimization_performed = True

    except ImportError:
        # geometric 없으면 berny fallback
        from pyscf.geomopt.berny_solver import optimize as berny_optimize
        opt_mol = berny_optimize(
            mf0,
            callback=_geomopt_callback,
            maxsteps=kwargs.get("max_steps", 100),
        )
        optimization_performed = True

    except Exception as exc:
        logger.warning("Geometry optimization failed: %s", exc)
        opt_mol = mol
        optimization_performed = False

    # ── 결과 조립 ──
    _emit_progress(progress_callback, 0.90, "optimize", "Building final result")

    final_result = _make_base_result(
        opt_mol, mf_final, method, basis, mol_input, **kwargs
    )
    final_result["optimization_performed"] = optimization_performed
    final_result["optimization_steps"] = len(trajectory)

    # ★ 궤적 데이터 삽입
    if trajectory:
        final_result["trajectory"] = trajectory

        # 멀티프레임 XYZ도 함께 제공 (3Dmol.js animate용)
        multi_xyz = _build_multi_frame_xyz(trajectory)
        if multi_xyz:
            final_result["trajectory_xyz"] = multi_xyz

    return final_result


def _mol_to_xyz_string(mol):
    """PySCF Mole 객체 → XYZ 포맷 문자열"""
    try:
        coords = mol.atom_coords(unit="Angstrom")
        symbols = [mol.atom_symbol(i) for i in range(mol.natm)]
        lines = [str(mol.natm), ""]
        for sym, (x, y, z) in zip(symbols, coords):
            lines.append(f"{sym}  {x: .8f}  {y: .8f}  {z: .8f}")
        return "\n".join(lines)
    except Exception:
        return None


def _build_multi_frame_xyz(trajectory):
    """trajectory 리스트 → 3Dmol.js가 읽을 수 있는 멀티프레임 XYZ 문자열.
    각 프레임은 표준 XYZ 포맷(원자수 / 코멘트 / 좌표들)이 연속된 형태."""
    frames = []
    for step in trajectory:
        xyz = step.get("xyz")
        if xyz:
            frames.append(xyz.strip())
    if not frames:
        return None
    return "\n".join(frames) + "\n"
```

### 3‑B. 프론트엔드 — `viewer.js` 트라젝토리 플레이어

```javascript
// ═══════════════════════════════════════════════════════════
// 요구사항 3: Geometry Optimization Trajectory 애니메이션
// ═══════════════════════════════════════════════════════════

/* ── 상태 확장 ── */
// state.trajectory 관련 속성 추가:
// state.trajectoryFrames = [];
// state.trajectoryPlaying = false;
// state.trajectoryFrame = 0;
// state.trajectoryTimer = null;

/**
 * 트라젝토리 데이터가 있으면 플레이어 UI를 초기화
 */
function initTrajectoryPlayer(result) {
  var $traj = document.getElementById("trajectoryPlayer");
  if (!$traj) return;

  var trajectory = result.trajectory;
  var multiXyz = result.trajectory_xyz;

  if (!trajectory || trajectory.length < 2) {
    $traj.hidden = true;
    state.trajectoryFrames = [];
    return;
  }

  /* 프레임 데이터 저장 */
  state.trajectoryFrames = trajectory;
  state.trajectoryFrame = trajectory.length - 1; // 최종 구조부터 표시
  state.trajectoryPlaying = false;

  /* UI 표시 */
  $traj.hidden = false;

  var $slider = document.getElementById("trajSlider");
  var $label = document.getElementById("trajLabel");
  var $btnPlay = document.getElementById("trajPlay");
  var $btnPause = document.getElementById("trajPause");

  if ($slider) {
    $slider.min = 0;
    $slider.max = trajectory.length - 1;
    $slider.value = $slider.max;
    $slider.addEventListener("input", function () {
      var idx = parseInt($slider.value, 10);
      goToTrajectoryFrame(idx);
    });
  }

  if ($btnPlay) {
    $btnPlay.addEventListener("click", function () {
      playTrajectory();
    });
  }

  if ($btnPause) {
    $btnPause.addEventListener("click", function () {
      pauseTrajectory();
    });
  }

  updateTrajectoryLabel();

  /* 대안: 3Dmol의 내장 멀티모델 애니메이션 사용 */
  if (multiXyz && state.viewer) {
    loadTrajectoryMultiModel(multiXyz);
  }
}

/**
 * 멀티프레임 XYZ를 3Dmol의 멀티모델로 로드
 */
function loadTrajectoryMultiModel(multiXyz) {
  if (!state.viewer) return;

  state.viewer.removeAllModels();
  state.viewer.addModelsAsFrames(multiXyz, "xyz");
  state.viewer.setStyle(
    {},
    {
      stick: { radius: 0.15 },
      sphere: { scale: 0.3 },
    },
  );
  state.viewer.zoomTo();
  state.viewer.render();

  /* 마지막 프레임 표시 */
  var nFrames = state.trajectoryFrames.length;
  if (nFrames > 0) {
    state.viewer.setFrame(nFrames - 1);
    state.viewer.render();
  }
}

/**
 * 특정 프레임으로 이동 (슬라이더 조작 시)
 */
function goToTrajectoryFrame(idx) {
  state.trajectoryFrame = idx;

  if (state.viewer) {
    /* 방법 1: 3Dmol 멀티모델 프레임 전환 */
    try {
      state.viewer.setFrame(idx);
      state.viewer.render();
    } catch (_) {
      /* 방법 2: 개별 XYZ 교체 (멀티모델이 없을 때) */
      var frame = state.trajectoryFrames[idx];
      if (frame && frame.xyz) {
        state.viewer.removeAllModels();
        state.viewer.addModel(frame.xyz, "xyz");
        state.viewer.setStyle(
          {},
          {
            stick: { radius: 0.15 },
            sphere: { scale: 0.3 },
          },
        );
        /* 트라젝토리 재생 중에는 zoomTo 하지 않음 */
        state.viewer.render();
      }
    }
  }

  var $slider = document.getElementById("trajSlider");
  if ($slider) $slider.value = idx;

  updateTrajectoryLabel();
}

/**
 * 라벨 업데이트 (Step N / Total, Energy)
 */
function updateTrajectoryLabel() {
  var $label = document.getElementById("trajLabel");
  if (!$label) return;

  var frames = state.trajectoryFrames;
  var idx = state.trajectoryFrame;
  var frame = frames[idx] || {};
  var total = frames.length;

  var text = "Step " + (idx + 1) + " / " + total;
  if (frame.energy_hartree != null) {
    text += "  |  E = " + frame.energy_hartree.toFixed(8) + " Ha";
  }
  if (frame.grad_norm != null) {
    text += "  |  |∇| = " + frame.grad_norm.toFixed(6);
  }

  $label.textContent = text;
}

/**
 * 재생: 첫 프레임부터 마지막까지 자동 전환
 */
function playTrajectory() {
  if (state.trajectoryPlaying) return;
  state.trajectoryPlaying = true;

  var frames = state.trajectoryFrames;
  if (!frames.length) return;

  /* 처음부터 재생하고 싶으면 0으로 리셋 */
  if (state.trajectoryFrame >= frames.length - 1) {
    state.trajectoryFrame = 0;
  }

  var $btnPlay = document.getElementById("trajPlay");
  var $btnPause = document.getElementById("trajPause");
  if ($btnPlay) $btnPlay.hidden = true;
  if ($btnPause) $btnPause.hidden = false;

  function nextFrame() {
    if (!state.trajectoryPlaying) return;
    if (state.trajectoryFrame >= frames.length - 1) {
      pauseTrajectory();
      return;
    }
    state.trajectoryFrame++;
    goToTrajectoryFrame(state.trajectoryFrame);
    state.trajectoryTimer = setTimeout(nextFrame, 200); // 200ms per frame
  }

  nextFrame();
}

/**
 * 일시정지
 */
function pauseTrajectory() {
  state.trajectoryPlaying = false;
  if (state.trajectoryTimer) {
    clearTimeout(state.trajectoryTimer);
    state.trajectoryTimer = null;
  }

  var $btnPlay = document.getElementById("trajPlay");
  var $btnPause = document.getElementById("trajPause");
  if ($btnPlay) $btnPlay.hidden = false;
  if ($btnPause) $btnPause.hidden = true;
}
```

## 요구사항 4: 멀티‑결과 세션 탭 UI

### 4‑A. results.js — 세션 결과 스택 + 탭 관리

```javascript
// ═══════════════════════════════════════════════════════════
// 요구사항 4: 단일 세션 내 다중 결과 탭
// ═══════════════════════════════════════════════════════════

/* ── 세션 결과 스택 ── */
var sessionResults = []; // { id, label, result, jobId, timestamp }
var activeSessionIdx = -1;

/**
 * 결과가 올 때마다 세션 스택에 push하고 탭 UI를 갱신
 * 기존 update() 함수의 첫 부분을 교체
 */
function update(result, jobId, source) {
  var normalized = normalizeResult(result);

  /* ── 세션 스택에 추가 ── */
  var label = buildResultLabel(normalized, sessionResults.length + 1);
  var entry = {
    id: jobId || "local-" + Date.now(),
    label: label,
    result: normalized,
    jobId: jobId,
    timestamp: Date.now(),
  };
  sessionResults.push(entry);
  activeSessionIdx = sessionResults.length - 1;

  /* ── 현재 활성 결과 설정 ── */
  state.result = normalized;
  state.jobId = jobId || null;

  /* ── 탭 UI 갱신 ── */
  renderSessionTabs();

  /* ── 이하 기존 렌더링 로직 동일 ── */
  renderResultPanels(normalized);
  App.emit("result:updated", { result: normalized, jobId: jobId });
}

/**
 * 결과 라벨 생성
 */
function buildResultLabel(result, index) {
  var mol = result.molecule_name || result.structure_name || "Mol";
  var type = "";

  if (result.optimization_performed) {
    type = "Opt";
  } else if (result.orbital_cube_b64 || result.orbital_cube) {
    /* 오비탈 라벨 추출 */
    var orbs = result.orbitals || [];
    var selIdx = 0; // 기본값
    for (var i = 0; i < orbs.length; i++) {
      if (orbs[i] && orbs[i].is_selected) {
        selIdx = i;
        break;
      }
    }
    var orbLabel = (orbs[selIdx] && orbs[selIdx].label) || "MO";
    type = orbLabel;
  } else if (result.esp_cube_b64 || result.esp_cube) {
    type = "ESP";
  } else {
    type = result.method || "SCF";
  }

  return "#" + index + " " + mol + " " + type;
}

/**
 * 세션 탭 바 렌더링
 */
function renderSessionTabs() {
  var $bar = document.getElementById("sessionTabBar");
  if (!$bar) return;

  $bar.innerHTML = "";
  $bar.hidden = sessionResults.length <= 1;

  for (var i = 0; i < sessionResults.length; i++) {
    var entry = sessionResults[i];
    var $tab = document.createElement("button");
    $tab.className = "session-tab" + (i === activeSessionIdx ? " active" : "");
    $tab.textContent = entry.label;
    $tab.title = new Date(entry.timestamp).toLocaleTimeString();
    $tab.setAttribute("data-idx", i);

    $tab.addEventListener(
      "click",
      (function (idx) {
        return function () {
          switchToSessionResult(idx);
        };
      })(i),
    );

    /* 닫기(X) 버튼 */
    var $close = document.createElement("span");
    $close.className = "session-tab-close";
    $close.textContent = "×";
    $close.title = "이 결과 닫기";
    $close.addEventListener(
      "click",
      (function (idx, evt) {
        return function (e) {
          e.stopPropagation();
          removeSessionResult(idx);
        };
      })(i),
    );

    $tab.appendChild($close);
    $bar.appendChild($tab);
  }
}

/**
 * 세션 탭 클릭 → 해당 결과로 전환
 */
function switchToSessionResult(idx) {
  if (idx < 0 || idx >= sessionResults.length) return;
  if (idx === activeSessionIdx) return;

  activeSessionIdx = idx;
  var entry = sessionResults[idx];

  state.result = entry.result;
  state.jobId = entry.jobId;

  renderSessionTabs();
  renderResultPanels(entry.result);

  /* 뷰어도 갱신 */
  App.emit("result:switched", { result: entry.result, jobId: entry.jobId });
}

/**
 * 세션 탭 닫기
 */
function removeSessionResult(idx) {
  if (idx < 0 || idx >= sessionResults.length) return;

  sessionResults.splice(idx, 1);

  if (sessionResults.length === 0) {
    activeSessionIdx = -1;
    state.result = null;
    state.jobId = null;
    renderSessionTabs();
    clearResultPanels();
    return;
  }

  /* 삭제된 탭이 활성이었으면 마지막 탭으로 이동 */
  if (idx === activeSessionIdx) {
    activeSessionIdx = Math.min(idx, sessionResults.length - 1);
    var entry = sessionResults[activeSessionIdx];
    state.result = entry.result;
    state.jobId = entry.jobId;
    renderResultPanels(entry.result);
  } else if (idx < activeSessionIdx) {
    activeSessionIdx--;
  }

  renderSessionTabs();
}

/**
 * 모든 결과 패널 클리어
 */
function clearResultPanels() {
  /* 각 탭 콘텐츠를 빈 상태로 */
  var ids = ["tabSummary", "tabOrbital", "tabESP", "tabCharges", "tabAdvisor"];
  ids.forEach(function (id) {
    var el = document.getElementById(id);
    if (el) el.innerHTML = "";
  });
  App.emit("result:cleared");
}
```

# QCViz‑MCP v5 — ESP 토글 버튼 먹통 수정

---

## 근본 원인 진단

이 문제는 세 가지가 겹쳐서 발생합니다.

첫째, **바인딩 시점 문제**입니다. `var $btnModeESP = document.getElementById("btnModeESP")`가 IIFE 최상단(또는 모듈 스코프 최상단)에서 실행되는데, 이 시점에 `index.html`의 `<button id="btnModeESP">`가 아직 파싱되지 않았으면 `null`이 반환됩니다. `null`에 `addEventListener`를 걸 수 없으므로 `if ($btnModeESP)` 가드에 의해 조용히 스킵됩니다.

둘째, **`switchVizMode("esp")` 내부의 silent failure**입니다. ESP cube 데이터가 `state.result`에 없거나 `atob()` 디코딩에서 에러가 나도 `try/catch`가 삼켜버려서, 버튼을 눌러도 화면에 아무 변화가 없습니다.

셋째, **`showControls` 연쇄 문제**입니다. 토글 버튼이 `hidden`인 채로 시작하는데, `showControls`가 제때 호출되지 않으면 버튼 자체가 보이지 않거나, 보여도 재클릭 시 다시 숨겨지는 경우가 있습니다.

## 수정된 viewer.js 코드

### 1단계: DOM 수집을 `init()` 안으로 이동

기존에 전역/IIFE 최상단에 흩어져 있던 `document.getElementById` 호출들을 `init()` → `collectDom()` 패턴으로 통합합니다.

```javascript
// ═══════════════════════════════════════════════════════════
// viewer.js — 수정된 초기화 구조
// ═══════════════════════════════════════════════════════════

(function (App) {
  "use strict";

  /* ── state (기존 그대로) ── */
  var state = {
    viewer: null,
    ready: false,
    mode: null, // "orbital" | "esp" | "structure"
    result: null,
    selectedOrbitalIndex: 0,
    isovalue: 0.03,
    opacity: 0.7,
    espDensityIso: 0.01,
    loadedXyzHash: null,
    // ... 기존 state 속성들 ...
  };

  /* ── DOM 참조 변수 (선언만, 할당은 collectDom에서) ── */
  var $viewerDiv;
  var $controls;
  var $empty;
  var $grpOrbital;
  var $grpESP;
  var $grpOpacity;
  var $selectOrbital;
  var $vizModeToggle;
  var $btnModeOrbital;
  var $btnModeESP;
  var $sliderIso;
  var $sliderOpacity;

  // ─────────────────────────────────────────────
  // collectDom: 모든 DOM 참조를 한 곳에서 수집
  // init()이 호출할 때 DOM이 이미 ready 상태
  // ─────────────────────────────────────────────
  function collectDom() {
    $viewerDiv =
      document.getElementById("viewer-container") ||
      document.getElementById("viewer");
    $controls = document.getElementById("viewer-controls");
    $empty = document.getElementById("viewer-empty");
    $grpOrbital = document.getElementById("grpOrbital");
    $grpESP = document.getElementById("grpESP");
    $grpOpacity = document.getElementById("grpOpacity");
    $selectOrbital = document.getElementById("selectOrbital");
    $vizModeToggle = document.getElementById("vizModeToggle");
    $btnModeOrbital = document.getElementById("btnModeOrbital");
    $btnModeESP = document.getElementById("btnModeESP");
    $sliderIso = document.getElementById("sliderIsovalue");
    $sliderOpacity = document.getElementById("sliderOpacity");

    /* ── 디버그: 누락 요소 경고 ── */
    var critical = {
      viewerDiv: $viewerDiv,
      btnModeOrbital: $btnModeOrbital,
      btnModeESP: $btnModeESP,
    };
    Object.keys(critical).forEach(function (key) {
      if (!critical[key]) {
        console.warn(
          "[viewer] collectDom: #" +
            key +
            " not found in DOM. " +
            "Check that index.html has this element.",
        );
      }
    });
  }

  // ─────────────────────────────────────────────
  // bindEvents: DOM 수집 후 이벤트 리스너 등록
  // ─────────────────────────────────────────────
  function bindEvents() {
    /* ── ESP / Orbital 토글 버튼 ── */
    if ($btnModeOrbital) {
      $btnModeOrbital.addEventListener("click", function () {
        console.log("[viewer] Orbital button clicked");
        switchVizMode("orbital");
      });
    } else {
      console.warn("[viewer] btnModeOrbital not found — binding skipped");
    }

    if ($btnModeESP) {
      $btnModeESP.addEventListener("click", function () {
        console.log("[viewer] ESP button clicked");
        switchVizMode("esp");
      });
    } else {
      console.warn("[viewer] btnModeESP not found — binding skipped");
    }

    /* ── 오비탈 셀렉터 ── */
    if ($selectOrbital) {
      $selectOrbital.addEventListener("change", function () {
        var idx = parseInt($selectOrbital.value, 10);
        if (isNaN(idx)) return;
        state.selectedOrbitalIndex = idx;
        App.emit("orbital:select", { orbital_index: idx });

        /* 캐시 우선, 없으면 silent fetch (이전 수정사항) */
        if (!tryRenderCachedOrbital(idx)) {
          silentFetchOrbital(idx);
        }
        saveViewerSnapshot();
      });
    }

    /* ── Isovalue 슬라이더 ── */
    if ($sliderIso) {
      $sliderIso.addEventListener("input", function () {
        state.isovalue = parseFloat($sliderIso.value) || 0.03;
        if (state.mode === "orbital") {
          reRenderCurrentSurface();
        }
      });
    }

    /* ── Opacity 슬라이더 ── */
    if ($sliderOpacity) {
      $sliderOpacity.addEventListener("input", function () {
        state.opacity = parseFloat($sliderOpacity.value) || 0.7;
        reRenderCurrentSurface();
      });
    }

    /* ── App 이벤트 ── */
    App.on("result:switched", onResultSwitched);
  }

  // ═══════════════════════════════════════════════════════════
  // init: 진입점 — DOMContentLoaded 이후에 호출됨
  // ═══════════════════════════════════════════════════════════
  function init() {
    collectDom();
    bindEvents();
    console.log("[viewer] Initialized. btnModeESP =", $btnModeESP);
  }

  /* ── App에 등록 ── */
  App.viewer = {
    init: init,
    getState: function () {
      return state;
    },
  };

  // ... (이하 함수 정의들) ...
})(window.App || (window.App = {}));
```

핵심은 `collectDom()`과 `bindEvents()`가 `init()` 안에서만 호출되고, `init()` 자체는 DOM이 준비된 후에 실행된다는 점입니다. `index.html` 하단 또는 `app.js`의 `DOMContentLoaded` 핸들러에서 `App.viewer.init()`을 호출하면 됩니다.

### 2단계: `switchVizMode` 강화 — 에러를 삼키지 않기

```javascript
// ═══════════════════════════════════════════════════════════
// switchVizMode: ESP/Orbital 전환 (에러 로깅 강화)
// ═══════════════════════════════════════════════════════════
function switchVizMode(newMode) {
  console.log(
    "[viewer] switchVizMode called:",
    newMode,
    "current:",
    state.mode,
  );

  if (!state.result) {
    console.warn("[viewer] switchVizMode: No result in state. Aborting.");
    return;
  }

  if (state.mode === newMode) {
    console.log("[viewer] switchVizMode: Already in mode", newMode);
    return;
  }

  var result = state.result;
  state.mode = newMode;

  /* 기존 표면 제거 (분자 모델 유지) */
  if (state.viewer) {
    state.viewer.removeAllSurfaces();
  }

  if (newMode === "orbital") {
    var orbCube = resolveOrbitalCube(result);
    if (!orbCube) {
      console.warn(
        "[viewer] switchVizMode('orbital'): " +
          "No orbital cube data found in result. " +
          "Keys available:",
        Object.keys(result).join(", "),
      );
      showControls(newMode);
      return;
    }
    renderOrbitalSurface(orbCube, state.isovalue, state.opacity);
  } else if (newMode === "esp") {
    var espData = resolveESPCube(result);
    if (!espData.esp) {
      console.warn(
        "[viewer] switchVizMode('esp'): " +
          "No ESP cube data found in result. " +
          "Keys available:",
        Object.keys(result).join(", "),
      );
      showControls(newMode);
      return;
    }
    renderESPSurface(espData.esp, espData.density, {
      density_iso: state.espDensityIso,
      esp_min: result.esp_min || result.esp_range_min,
      esp_max: result.esp_max || result.esp_range_max,
    });
  } else {
    /* structure 모드: 표면 없이 분자만 */
    if (state.viewer) state.viewer.render();
  }

  showControls(newMode);
  saveViewerSnapshot();
}

// ─────────────────────────────────────────────
// 큐브 데이터 해석 헬퍼: 여러 필드명 후보를 시도
// ─────────────────────────────────────────────
function resolveOrbitalCube(result) {
  /* 이미 디코딩된 문자열 */
  if (result.orbital_cube) return result.orbital_cube;

  /* base64 인코딩 */
  if (result.orbital_cube_b64) {
    try {
      return atob(result.orbital_cube_b64);
    } catch (e) {
      console.error("[viewer] atob failed for orbital_cube_b64:", e);
      return null;
    }
  }

  /* 선택된 오비탈의 개별 cube */
  var orbs = result.orbitals || [];
  var idx = state.selectedOrbitalIndex || 0;
  var orb = orbs[idx];
  if (orb) {
    if (orb.cube_data) return orb.cube_data;
    if (orb.cube_b64) {
      try {
        return atob(orb.cube_b64);
      } catch (e) {
        console.error("[viewer] atob failed for orb.cube_b64:", e);
      }
    }
  }

  /* visualization 객체 내 */
  var viz = result.visualization || {};
  if (viz.orbital_cube) return viz.orbital_cube;
  if (viz.orbital_cube_b64) {
    try {
      return atob(viz.orbital_cube_b64);
    } catch (e) {
      console.error("[viewer] atob failed for viz.orbital_cube_b64:", e);
    }
  }

  return null;
}

function resolveESPCube(result) {
  var esp = null;
  var density = null;

  /* ── ESP cube ── */
  if (result.esp_cube) {
    esp = result.esp_cube;
  } else if (result.esp_cube_b64) {
    try {
      esp = atob(result.esp_cube_b64);
    } catch (e) {
      console.error("[viewer] atob failed for esp_cube_b64:", e);
    }
  }

  /* visualization 객체 내 */
  if (!esp) {
    var viz = result.visualization || {};
    if (viz.esp_cube) {
      esp = viz.esp_cube;
    } else if (viz.esp_cube_b64) {
      try {
        esp = atob(viz.esp_cube_b64);
      } catch (e) {
        console.error("[viewer] atob failed for viz.esp_cube_b64:", e);
      }
    }
  }

  /* ── Density cube (ESP 착색용) ── */
  if (result.density_cube) {
    density = result.density_cube;
  } else if (result.density_cube_b64) {
    try {
      density = atob(result.density_cube_b64);
    } catch (e) {
      /* density 없어도 ESP 단독 렌더 가능 */
    }
  }

  return { esp: esp, density: density };
}
```

`resolveOrbitalCube`와 `resolveESPCube`는 백엔드가 데이터를 어떤 필드명으로 보내든 찾아내는 "방어적 탐색기"입니다. 이전에 `switchVizMode`가 `result.esp_cube_b64` 하나만 확인하고 없으면 조용히 `return`하던 것을, 4~5개 후보 경로를 모두 탐색하고 최종적으로 없을 때만 `console.warn`을 남기도록 바꿨습니다.

## 요구사항 2: 표면 중첩 버그 수정

### 문제 재현 경로

Orbital → ESP → Orbital → ESP 를 빠르게 반복하면, `removeAllSurfaces()` 호출과 `addIsosurface()` 사이에 이전 비동기 렌더가 끼어들거나, `render()` 전에 이전 프레임의 표면이 남습니다. 3Dmol.js의 `removeAllSurfaces()`는 동기이지만, 연속 호출 시 내부 surface 배열 정리와 WebGL 버퍼 해제 타이밍이 어긋날 수 있습니다.

### viewer.js — 강화된 `switchVizMode`

```javascript
// ═══════════════════════════════════════════════════════════
// 요구사항 2: 표면 중첩 방지 — 확실한 클린업
// ═══════════════════════════════════════════════════════════

/* 전환 잠금: 빠른 연속 클릭 시 이전 전환이 끝나기 전에 새 전환 방지 */
var _vizSwitchLock = false;

function switchVizMode(newMode) {
  console.log(
    "[viewer] switchVizMode:",
    newMode,
    "current:",
    state.mode,
    "locked:",
    _vizSwitchLock,
  );

  if (!state.result) {
    console.warn("[viewer] switchVizMode: No result in state.");
    return;
  }
  if (state.mode === newMode) return;
  if (_vizSwitchLock) {
    console.log("[viewer] switchVizMode: Locked — ignoring rapid click");
    return;
  }

  _vizSwitchLock = true;
  state.mode = newMode;

  /* ── STEP 1: 기존 표면 완전 제거 ── */
  if (state.viewer) {
    /*
     * removeAllSurfaces만으로 부족한 경우를 대비:
     * removeAllShapes도 호출하여 혹시 shape으로 추가된
     * 시각 요소도 정리
     */
    state.viewer.removeAllSurfaces();
    if (typeof state.viewer.removeAllShapes === "function") {
      state.viewer.removeAllShapes();
    }
    /* 즉시 한 번 렌더하여 GPU 버퍼를 플러시 */
    state.viewer.render();
  }

  /* ── STEP 2: 새 표면 추가 (약간의 지연으로 클린업 보장) ── */
  setTimeout(function () {
    try {
      if (newMode === "orbital") {
        addOrbitalSurfaces();
      } else if (newMode === "esp") {
        addESPSurface();
      }
      if (state.viewer) state.viewer.render();
    } catch (e) {
      console.error("[viewer] Surface rendering error:", e);
    }

    showControls(newMode);
    saveViewerSnapshot();
    _vizSwitchLock = false;
  }, 30);
  /* 30ms: 한 프레임(16ms) 이상 대기하여 WebGL이 이전 surface를
     확실히 해제한 뒤 새 surface를 추가 */
}

/**
 * 오비탈 표면 추가 (switchVizMode에서 분리)
 */
function addOrbitalSurfaces() {
  var cubeData = resolveOrbitalCube(state.result);
  if (!cubeData) {
    console.warn("[viewer] addOrbitalSurfaces: No cube data");
    return;
  }

  var scheme = getCurrentColorScheme();
  var vol = new window.$3Dmol.VolumeData(cubeData, "cube");

  /* 양의 로브 */
  state.viewer.addIsosurface(vol, {
    isoval: state.isovalue,
    color: scheme.orbPositive,
    alpha: state.opacity,
    smoothness: 2,
  });

  /* 음의 로브 */
  state.viewer.addIsosurface(vol, {
    isoval: -state.isovalue,
    color: scheme.orbNegative,
    alpha: state.opacity,
    smoothness: 2,
  });
}

/**
 * ESP 표면 추가 (switchVizMode에서 분리)
 */
function addESPSurface() {
  var espData = resolveESPCube(state.result);
  if (!espData.esp) {
    console.warn("[viewer] addESPSurface: No ESP cube data");
    return;
  }

  var scheme = getCurrentColorScheme();
  var espVol = new window.$3Dmol.VolumeData(espData.esp, "cube");
  var densVol = espData.density
    ? new window.$3Dmol.VolumeData(espData.density, "cube")
    : espVol;

  var range = state.result.esp_max || state.result.esp_range_max || 0.05;

  state.viewer.addIsosurface(densVol, {
    isoval: state.espDensityIso || 0.001,
    color: "white",
    alpha: state.opacity,
    smoothness: 1,
    voldata: espVol,
    volscheme: scheme.espGradient(-range, range),
  });
}
```

핵심 변경은 두 가지입니다. 첫째, `removeAllSurfaces()` 후 즉시 `render()`를 한 번 호출하여 GPU 버퍼를 강제 플러시합니다. 둘째, 새 표면 추가를 `setTimeout(fn, 30)`으로 다음 프레임으로 미뤄, 클린업과 추가가 같은 렌더 사이클에 겹치지 않도록 합니다. `_vizSwitchLock`은 이 30ms 창 안에 사용자가 다시 클릭하는 것을 방지합니다.

## 요구사항 1: Color Scheme 변경 시 즉각 반영

### 문제 근본 원인 (3가지)

코드를 전수 분석한 결과 문제는 세 곳에서 동시에 발생하고 있습니다.

**원인 A — `renderOrbital`에서 음의 로브 색상 하드코딩.** `renderOrbital` 함수 내부에서 양의 로브는 `getCurrentColorScheme().orbPositive`를 쓰지만, 음의 로브는 `"#f59e0b"`로 하드코딩되어 있습니다. 스킴을 바꿔도 음의 로브는 항상 amber색입니다.

```javascript
// renderOrbital 내부 — 현재 코드
viewer.addIsosurface(vol, {
  isoval: state.isovalue,
  color: getCurrentColorScheme().orbPositive, // ← 스킴 반영 ✓
  // ...
});
viewer.addIsosurface(vol, {
  isoval: -state.isovalue,
  color: "#f59e0b", // ← 하드코딩 ✗ — 스킴 무시
  // ...
});
```

**원인 B — `init()`의 Color Scheme change 핸들러가 `_vizSwitchLock`에 막힘.** `state.mode = "none"` → `switchVizMode(currentMode)` 경로로 재렌더를 시도하는데, `switchVizMode` 첫 줄의 `state.mode === newMode` 체크는 통과하지만, `_vizSwitchLock`이 `true`인 채로 남아있을 수 있습니다(30ms timeout 내 연속 클릭 시). 또한 `switchVizMode`는 `setTimeout(fn, 30)` 안에서 렌더링하므로, 그 30ms 사이에 lock 상태가 변할 수 있습니다.

**원인 C — `renderOrbital`/`renderESP`가 `switchVizMode`와 별도 경로.** Isovalue/Opacity 슬라이더의 `change` 이벤트는 `renderOrbital(state.result)` / `renderESP(state.result)`를 직접 호출합니다. 이 함수들은 `clearViewer` → `addMoleculeModel` → `addIsosurface`를 처음부터 다시 하는데, Color Scheme 변경도 이 경로를 타야 확실합니다. 하지만 현재 Color Scheme change 핸들러는 `switchVizMode`를 타고 있어서, 두 경로의 렌더링 로직이 일치하지 않습니다.

## 요구사항 2: 원자 라벨에 전하량별 색상 부여

### 설계

Mulliken 전하를 기준으로 양수(+)면 파란색 계열, 음수(-)면 빨간색 계열, 0에 가까우면 기존 중성 색상을 적용합니다. 색의 강도는 전하량의 절대값에 비례합니다.

위치: `viewer.js`의 기존 `addLabels` 함수를 전체 교체

```javascript
// ═══ AFTER ═══
function addLabels(viewer, result) {
  var atoms = result.atoms || [];
  if (!atoms.length) return;

  var isDark = document.documentElement.getAttribute("data-theme") === "dark";

  /* ── 전하 배열 수집 (Mulliken 우선, Löwdin 폴백) ── */
  var charges = result.mulliken_charges || result.lowdin_charges || [];

  /* 전하 최대 절대값 (색상 정규화용) */
  var maxAbs = 0;
  for (var k = 0; k < charges.length; k++) {
    var cv = charges[k];
    var cval = cv != null && typeof cv === "object" ? cv.charge : cv;
    if (cval != null && isFinite(cval)) {
      var abs = Math.abs(cval);
      if (abs > maxAbs) maxAbs = abs;
    }
  }
  if (maxAbs < 0.001) maxAbs = 1; /* 전하가 거의 0이면 정규화 비활성 */

  atoms.forEach(function (a, i) {
    var el = a.element || a.symbol || a[0] || "";
    if (!el) return;

    /* 이 원자의 전하값 */
    var rawCharge = charges[i];
    var chargeVal = null;
    if (rawCharge != null) {
      chargeVal = typeof rawCharge === "object" ? rawCharge.charge : rawCharge;
      if (chargeVal != null && !isFinite(chargeVal)) chargeVal = null;
    }

    /* ── 라벨 텍스트: 원소 + 전하값 ── */
    var labelText = el;
    if (chargeVal != null) {
      var sign = chargeVal >= 0 ? "+" : "";
      labelText += " (" + sign + chargeVal.toFixed(3) + ")";
    }

    /* ── 전하 기반 색상 계산 ── */
    var bgColor, fontColor, borderColor;

    if (chargeVal != null && Math.abs(chargeVal) > 0.005) {
      /* 정규화된 강도 (0~1) */
      var intensity = Math.min(Math.abs(chargeVal) / maxAbs, 1.0);
      /* 최소 투명도 보장 */
      var alpha = 0.25 + intensity * 0.55; /* 0.25 ~ 0.80 */

      if (chargeVal > 0) {
        /* 양수 → 파란색 계열 */
        bgColor = "rgba(59, 130, 246, " + alpha.toFixed(2) + ")";
        fontColor = isDark ? "#dbeafe" : "#1e3a5f";
        borderColor = "rgba(59, 130, 246, 0.4)";
      } else {
        /* 음수 → 빨간색 계열 */
        bgColor = "rgba(239, 68, 68, " + alpha.toFixed(2) + ")";
        fontColor = isDark ? "#fee2e2" : "#7f1d1d";
        borderColor = "rgba(239, 68, 68, 0.4)";
      }
    } else {
      /* 중성 또는 전하 데이터 없음 → 기존 색상 */
      fontColor = isDark ? "white" : "#333";
      bgColor = isDark ? "rgba(0,0,0,0.5)" : "rgba(255,255,255,0.7)";
      borderColor = isDark ? "rgba(255,255,255,0.1)" : "rgba(0,0,0,0.1)";
    }

    viewer.addLabel(labelText, {
      position: {
        x: a.x != null ? a.x : a[1] || 0,
        y: a.y != null ? a.y : a[2] || 0,
        z: a.z != null ? a.z : a[3] || 0,
      },
      fontSize: 11,
      fontColor: fontColor,
      backgroundColor: bgColor,
      borderColor: borderColor,
      borderThickness: 1,
      backgroundOpacity: 0.85,
      alignment: "center",
      showBackground: true,
    });
  });
}
```

### results.js — `renderCharges` 전체 교체

위치: `results.js`에서 기존 `function renderCharges(r) { ... }` 블록 전체를 아래로 교체

```javascript
// ═══ BEFORE ═══
// (기존 단순 테이블 렌더링)

// ═══ AFTER ═══
function renderCharges(r) {
  var mull = r.mulliken_charges || [];
  var lowd = r.lowdin_charges || [];
  var atoms = r.atoms || [];
  if (!mull.length && !lowd.length)
    return '<p class="result-note">No charge data.</p>';

  /* ── 어떤 전하 세트를 차트에 쓸지 결정 ── */
  var primary = mull.length ? mull : lowd;
  var secondary = mull.length && lowd.length ? lowd : null;
  var primaryLabel = mull.length ? "Mulliken" : "L\u00F6wdin";
  var secondaryLabel = secondary ? "L\u00F6wdin" : null;

  /* 전하값 추출 헬퍼 */
  function chargeVal(arr, i) {
    var v = arr[i];
    if (v == null) return null;
    if (typeof v === "object")
      return v.charge != null ? Number(v.charge) : null;
    return Number(v);
  }

  /* ── 최대 절대값 계산 (차트 스케일용) ── */
  var maxAbs = 0;
  var n = Math.max(primary.length, secondary ? secondary.length : 0);
  for (var i = 0; i < n; i++) {
    var pv = chargeVal(primary, i);
    if (pv != null && Math.abs(pv) > maxAbs) maxAbs = Math.abs(pv);
    if (secondary) {
      var sv = chargeVal(secondary, i);
      if (sv != null && Math.abs(sv) > maxAbs) maxAbs = Math.abs(sv);
    }
  }
  if (maxAbs < 0.0001) maxAbs = 0.5; /* 전부 0에 가까우면 기본 스케일 */

  /* ── HTML 생성 ── */
  var html = "";

  /* 범례 */
  html += '<div class="butterfly-legend">';
  html +=
    '<span class="butterfly-legend__item">' +
    '<span class="butterfly-legend__swatch butterfly-legend__swatch--neg"></span>' +
    "Negative (\u2212)</span>";
  html +=
    '<span class="butterfly-legend__item">' +
    '<span class="butterfly-legend__swatch butterfly-legend__swatch--pos"></span>' +
    "Positive (+)</span>";
  if (secondary) {
    html +=
      '<span class="butterfly-legend__item" style="margin-left:auto;">' +
      '<span style="font-size:11px;color:var(--text-3);">' +
      '<span style="display:inline-block;width:8px;height:8px;border-radius:50%;background:var(--accent);margin-right:4px;"></span>' +
      primaryLabel +
      ' &nbsp; <span style="display:inline-block;width:8px;height:8px;border-radius:50%;background:var(--accent-2);margin-right:4px;"></span>' +
      secondaryLabel +
      "</span></span>";
  }
  html += "</div>";

  /* 차트 행들 */
  html += '<div class="butterfly-chart">';

  for (var i = 0; i < n; i++) {
    var el = atoms[i]
      ? atoms[i].element || atoms[i].symbol || atoms[i][0] || "?"
      : "?";

    var pv = chargeVal(primary, i);
    var sv = secondary ? chargeVal(secondary, i) : null;

    html += '<div class="butterfly-row">';

    /* ── 왼쪽: 음수 막대 영역 ── */
    html += '<div class="butterfly-bar-area butterfly-bar-area--neg">';
    if (pv != null && pv < 0) {
      var pctP = ((Math.abs(pv) / maxAbs) * 100).toFixed(1);
      html +=
        '<div class="butterfly-bar butterfly-bar--neg-primary" ' +
        'style="width:' +
        pctP +
        '%" ' +
        'title="' +
        primaryLabel +
        ": " +
        pv.toFixed(6) +
        '">' +
        '<span class="butterfly-bar__val">' +
        pv.toFixed(4) +
        "</span>" +
        "</div>";
    }
    if (sv != null && sv < 0) {
      var pctS = ((Math.abs(sv) / maxAbs) * 100).toFixed(1);
      html +=
        '<div class="butterfly-bar butterfly-bar--neg-secondary" ' +
        'style="width:' +
        pctS +
        '%" ' +
        'title="' +
        secondaryLabel +
        ": " +
        sv.toFixed(6) +
        '"></div>';
    }
    html += "</div>";

    /* ── 중앙: 원자 라벨 ── */
    html +=
      '<div class="butterfly-label">' +
      '<span class="butterfly-label__idx">' +
      (i + 1) +
      "</span>" +
      '<span class="butterfly-label__el">' +
      esc(el) +
      "</span>" +
      "</div>";

    /* ── 오른쪽: 양수 막대 영역 ── */
    html += '<div class="butterfly-bar-area butterfly-bar-area--pos">';
    if (pv != null && pv >= 0) {
      var pctP = ((Math.abs(pv) / maxAbs) * 100).toFixed(1);
      html +=
        '<div class="butterfly-bar butterfly-bar--pos-primary" ' +
        'style="width:' +
        pctP +
        '%" ' +
        'title="' +
        primaryLabel +
        ": " +
        (pv >= 0 ? "+" : "") +
        pv.toFixed(6) +
        '">' +
        '<span class="butterfly-bar__val">+' +
        pv.toFixed(4) +
        "</span>" +
        "</div>";
    }
    if (sv != null && sv >= 0) {
      var pctS = ((Math.abs(sv) / maxAbs) * 100).toFixed(1);
      html +=
        '<div class="butterfly-bar butterfly-bar--pos-secondary" ' +
        'style="width:' +
        pctS +
        '%" ' +
        'title="' +
        secondaryLabel +
        ": +" +
        sv.toFixed(6) +
        '"></div>';
    }
    html += "</div>";

    html += "</div>"; /* .butterfly-row */
  }

  html += "</div>"; /* .butterfly-chart */

  /* ── 하단에 접이식 상세 테이블도 유지 ── */
  html += '<details style="margin-top:var(--sp-4)">';
  html += "<summary>Detailed Charge Table</summary>";
  html +=
    '<table class="result-table" style="margin-top:var(--sp-2)"><thead><tr>' +
    "<th>#</th><th>Element</th>";
  if (mull.length) html += "<th>Mulliken</th>";
  if (lowd.length) html += "<th>L\u00F6wdin</th>";
  html += "</tr></thead><tbody>";
  for (var i = 0; i < n; i++) {
    var el = atoms[i]
      ? atoms[i].element || atoms[i].symbol || atoms[i][0] || "?"
      : "?";
    html += "<tr><td>" + (i + 1) + "</td><td>" + esc(el) + "</td>";
    if (mull.length) {
      var mv = chargeVal(mull, i);
      html += "<td>" + (mv != null ? mv.toFixed(6) : "\u2014") + "</td>";
    }
    if (lowd.length) {
      var lv = chargeVal(lowd, i);
      html += "<td>" + (lv != null ? lv.toFixed(6) : "\u2014") + "</td>";
    }
    html += "</tr>";
  }
  html += "</tbody></table></details>";

  return html;
}
```

## 적용 후 검증

첫째, Color Scheme 셀렉터에서 "Viridis"를 선택합니다. 오비탈의 양의 로브가 `#21918c`(청록), 음의 로브가 `#fde725`(노랑)으로 즉시 바뀌는지 확인합니다. Legend의 스와치도 동일한 색상이어야 합니다. 이전에 문제였던 "음의 로브가 항상 amber"가 사라졌는지 특히 주의 깊게 봅니다.

둘째, Labels 버튼을 켠 상태에서 전하 데이터가 있는 분자를 봅니다. 양의 전하를 가진 원자(예: H)의 라벨 배경이 파란색 계열이고, 음의 전하를 가진 원자(예: O)의 라벨 배경이 빨간색 계열인지 확인합니다. 전하값이 0에 가까운 원자는 기존의 반투명 흰색/검은색 배경이어야 합니다.

셋째, Results 패널의 Charges 탭을 엽니다. 0을 중심으로 왼쪽(빨강)과 오른쪽(파랑)으로 뻗어나가는 수평 막대가 각 원자별로 표시되는지 확인합니다. Mulliken과 Löwdin 전하가 모두 있으면 두 번째(얇은) 막대가 겹쳐 표시됩니다. 하단의 "Detailed Charge Table" 접기를 펼치면 기존 숫자 테이블도 여전히 볼 수 있어야 합니다.
